# Hunting grounds line for the external estate slots.
# Warren -> Deer Park, then it forks: The Chase (prestige and hunting) or
# Warren & Fishponds (income and quiet living). Same fork setup as the dungeon line.
# Names are the real terms btw: a warren was the right to take small game, a park was
# an enclosed deer ground (you needed a royal license to empark in England), and a
# chase was a magnate's larger hunting territory.
#
# Art note: using the empty grassland slot texture as a stand in for a forest until
# we have proper art, so there's just one asset block per building instead of the
# usual per culture set.

estex_warren_01 = {
	construction_time = 365
	allowed_domicile_types = { estex_estate }

	cost = {
		gold = estate_external_building_high_cost_tier_1_value
	}

	character_modifier = {
		domicile_monthly_gold_add = 0.1
		stress_gain_mult = -0.05
	}

	parameters = {
		estex_estate_has_hunting_grounds = yes
	}

	ai_value = {
		value = 8
		if = {
			limit = {
				scope:owner.culture ?= {
					has_cultural_tradition = tradition_hunters
				}
			}
			add = 2
		}
		add = estate_building_ai_modifier_value
	}

	asset = {
		icon = "gfx/interface/icons/domicile_building/domicile_hunt.dds"
		texture = "gfx/interface/window_domiciles/estate_jungle_empty_03.dds"
		intersectionmask_texture = "gfx/interface/window_domiciles/estate_jungle_empty_03_mask.png"
		soundeffect = "event:/DLC/EP3/SFX/UI/domicile_buildings/ep3_ui_domicile_buildings_garden_fruit"
	}
}

estex_deer_park_02 = {
	construction_time = 730

# Prestige and hunting path. The kennels_more_successful_hunts_1 parameter is a real
# vanilla one (the estate kennels use it) so hunts actually get better without us
# writing any activity script.
estex_chase_03 = {
	construction_time = 730
	allowed_domicile_types = { estex_estate }
	previous_building = estex_deer_park_02

	cost = {
		gold = estate_external_building_high_cost_tier_3_value
	}

	character_modifier = {
		monthly_prestige = 0.3
		prowess = 1
		stress_gain_mult = -0.1
	}

	parameters = {
		estex_estate_has_hunting_grounds = yes
		estex_estate_has_deer_park = yes
		estex_estate_has_chase = yes
		kennels_more_successful_hunts_1 = yes
	}

	ai_value = {
		value = 7
		if = {
			limit = {
				scope:owner.culture ?= {
					has_cultural_tradition = tradition_hunters
				}
			}
			add = 3
		}
		add = estate_building_ai_modifier_value
	}

	asset = {
		icon = "gfx/interface/icons/domicile_building/domicile_building_trackers.dds"
		texture = "gfx/interface/window_domiciles/estate_jungle_empty_03.dds"
		intersectionmask_texture = "gfx/interface/window_domiciles/estate_jungle_empty_03_mask.png"
		soundeffect = "event:/DLC/EP3/SFX/UI/domicile_buildings/ep3_ui_domicile_buildings_garden_fruit"
	}
}
