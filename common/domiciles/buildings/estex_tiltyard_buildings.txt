# Tournament line for the external estate slots.
# Tournament Grounds -> Berfrois (the raised stands where the lord watches, real
# medieval term), then the usual fork: Jousting Lists (knights and glory) or
# Archery Butts (levies and the common soldiery). Chivalry cultures lean jousting,
# longbow cultures lean archery, everyone else picks on taste.
#
# Art note: placeholder textures again. Horse pastures (only has the japanese
# variant but it's a fenced riding ground, looks fine) for the mounted side,
# grazing fields for the archery side.

estex_tournament_grounds_01 = {
	construction_time = 365
	allowed_domicile_types = { estex_estate }

	cost = {
		gold = estate_external_building_high_cost_tier_1_value
	}

	character_modifier = {
		monthly_martial_lifestyle_xp_gain_mult = 0.1
		knight_effectiveness_mult = 0.05
	}

	parameters = {
		estex_estate_has_tournament_grounds = yes
	}

	ai_value = {
		value = 8
		if = {
			limit = {
				scope:owner.culture ?= {
					has_cultural_tradition = tradition_chivalry
				}
			}
			add = 2
		}
		add = estate_building_ai_modifier_value
	}

	asset = {
		icon = "gfx/interface/icons/domicile_building/domicile_building_horse_run.dds"
		texture = "gfx/interface/window_domiciles/laamp_building_proving_grounds.dds"
		intersectionmask_texture = "gfx/interface/window_domiciles/laamp_building_proving_grounds_mask.png"
		soundeffect = "event:/DLC/EP3/SFX/UI/domicile_buildings/ep3_ui_domicile_buildings_guardhouse"
	}
}
